/** * Gestione dinamica degli articoli per Data Strategy Consulting */ async function loadArticles() { const container = document.getElementById('articles-container'); try { // Carica i dati dal file JSON nella root del repository const response = await fetch('articles.json'); if (!response.ok) { throw new Error('Impossibile caricare il database articoli'); } const articles = await response.json(); // Pulisce i loader (skeleton screens) container.innerHTML = ''; articles.forEach(article => { const articleElement = createArticleCard(article); container.appendChild(articleElement); }); } catch (error) { console.error("Errore critico:", error); container.innerHTML = `

Le nostre ultime analisi sono in fase di revisione. Torna a trovarci a breve.

`; } } /** * Crea l'elemento HTML per un singolo articolo */ function createArticleCard(article) { const wrapper = document.createElement('article'); wrapper.className = 'group cursor-pointer hover-lift'; wrapper.innerHTML = `
${article.category}
§
${article.date}

${article.title}

${article.abstract}

Leggi PDF (LaTeX) ${article.discussion ? `Discussione` : ''}
`; // Aggiungiamo classi specifiche per i link tramite JS per mantenere l'HTML pulito wrapper.querySelectorAll('.read-more-link').forEach(link => { link.className = 'text-[10px] font-bold mck-blue uppercase tracking-widest border-b border-blue-900 pb-1 hover:text-blue-700 transition-colors'; }); if (article.discussion) { wrapper.querySelector('.discussion-link').className = 'text-[10px] font-bold text-gray-400 uppercase tracking-widest border-b border-gray-200 pb-1 hover:text-gray-600 transition-colors'; } return wrapper; } // Inizializza al caricamento del DOM document.addEventListener('DOMContentLoaded', loadArticles);
